    table_name = to_upper( io_rap_bo_node->data_source_name ).

    DATA(xco_library) = NEW zdmo_cl_rap_xco_on_prem_lib( ).

    IF xco_library->on_premise_branch_is_used(  ).

        table_name             = table_name
